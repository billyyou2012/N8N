SELECT 
    SPECIFIC_NAME as procedure_name,
    ROUTINE_TYPE as procedure_type
FROM information_schema.routines
WHERE SPECIFIC_NAME LIKE '%insert_schedule%' 
   OR SPECIFIC_NAME LIKE '%schedule%';
